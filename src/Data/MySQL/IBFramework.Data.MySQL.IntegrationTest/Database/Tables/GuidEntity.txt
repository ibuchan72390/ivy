CREATE TABLE `guidentity` (
  `Id` varchar(40) NOT NULL,
  `Name` varchar(200) DEFAULT NULL,
  `Integer` int(11) DEFAULT NULL,
  `Decimal` decimal(10,2) DEFAULT NULL,
  `Double` double DEFAULT NULL,
  `CoreEntityId` int(11) DEFAULT NULL,
